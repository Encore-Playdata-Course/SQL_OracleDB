select ((countnov - countoct)*100/countoct)  from place;

30 % 이상인 데이터가 두 개네
경찰 증원을 1.3배고 인원 수를 나타내야하니 반올림해줘야겠네

select loc, round(countpolice*1.3) as countpolice
from (select trunc((countnov-countoct)*100/countoct) as inc, countpolice, loc
from place)
where inc >= 30;
