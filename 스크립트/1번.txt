people테이블에는 어떤 게 있지??

select case, name, persontype from people;

동일한 케이스 번호를 공유하는 피해자, 가해자가 다수인 데이터가 존재하네
그럼 테이블 조인을 한 후 6번째 값을 가져온다면??

select rn, case, persontype, time 
from (select row_number() over (order by b.time asc, b.case)  as rn, b.case, b.time, a.persontype
from people a, crime b 
where a.case = b.case)
where rn = 6;

데이터가 하나밖에 안 나오네...
당연하지... 바보야... 케이스를 중심으로 검색해야 답이 나오지

select name, persontype
from people, crime
where people.case = crime.case and people.case = 10000;

이 사건이 정답이구나 ^^
혹시 ROW_NUMBER()라는 함수로 풀 수 있을까??

select rn, uniquecase, persontype, name, time 
from people, (select row_number() over (order by time asc, case)  as rn, case as uniquecase, time from crime) 
where rn = 6 and uniquecase = case;
